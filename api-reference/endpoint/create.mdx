### Body

<ResponseField name="video_url" type="string" required="true">
  Video URL to lipsync
</ResponseField>

<ResponseField name="audio_url" type="string">
  Audio URL to lipsync
</ResponseField>

<ResponseField name="script" type="object">  
  <Expandable title="Toggle script child">
    
    <ResponseField name="text" type="string">
      Text to speak
    </ResponseField>
    
    <ResponseField name="voice_id" type="string">
      Voice ID
    </ResponseField>
  
  </Expandable>
</ResponseField>

### Response

<ResponseField name="id" type="string">
  Video generation job id
</ResponseField>

<RequestExample>

```bash Bash
curl --location --request POST 'https://pulse.personate.ai/api/express_lipsync_video' \
--header 'Content-Type: application/json' \
--header 'Authorization: Token <token>' \
--data '{
    "video_url": "",
    "audio_url": "",
    "script": {
      "text": "",
      "voice_id": ""
    }
}'
```

```python Python
import requests

url = "https://pulse.personate.ai/api/express_lipsync_video"

payload = {
    "video_url": "",
    "audio_url": "",
    "script": {
      "text": "",
      "voice_id": ""
    }
}
headers = {
    "Authorization": "Bearer <api-key>",
    "Content-Type": "application/json"
}

response = requests.request("POST", url, json=payload, headers=headers)

print(response.text)
```

```python Python
import requests
import json

url = "https://pulse.personate.ai/api/agentic_job"

payload = json.dumps({
  "video_name": "string",
  "template_id": "string",
  "webhook": "string",
  "agent_steps": [
    "step_name": "string",
    "step_key": "string",
    "step_input": "object"
  ]
})
headers = {
  'Content-Type': 'application/json',
  'Authorization': 'Bearer ********'
}

response = requests.request("POST", url, headers=headers, data=payload)

print(response.text)
```
